## 变量内存地址分析

```go
//
// Created by virus on 2022/5/20.
//

#include "stdio.h"

int main() {
    // <type> <name>
    int value; // 不赋值默认为0

    // <type> <name> = <initialized value>
    int value_init = 3;

    value = 4;
    value_init = 5;

    printf("value :%d\n", value);

    value_init = value;
    printf("size of value: %d\n", sizeof(value)); // 打印字节数

    printf("address of value: %#x\n", &value); // 打印十六进制内存地址
    return 0;
}
```



## 访问内存上的数据指令

### 通过虚拟地址别名【变量名、函数名】访问内存上的数据和指令

```c
#include <stdio.h>
#include <stdlib.h>

int x = 10;

int test()
{
        return x;
}

int main()
{
        printf("x=%d\r\n", x);
        printf("x=%p\r\n", &x);

        printf("test=%p\r\n", test);

        char *test_text = (char*)(test);

        int i = 0;
        for (i;i < 12; i++) {
                printf("%x\r\n", test_text[i]);
        }
        return 0;
}
```

## 通过虚拟地址访问地址上的数据



```c
int main()
{
        printf("x=%d\r\n", x);
        printf("x=%p\r\n", &x);

        printf("test=%p\r\n", test);
        printf("=====================\r\n");
        char *test_text = (char*)(test);

        int i = 0;
        for (i;i < 12; i++) {
                printf("%x\r\n", test_text[i]);
        }
        printf("=================\r\n");
        printf("%d\r\n", *(int*)(0x60103d));
        printf("=================\r\n");
        char j = 0;
        for (j = 0; j < 12 ;j++) {
                printf("%x\r\n", *(char*)(0x40057d+j));
        }
        return 0;
}
```
