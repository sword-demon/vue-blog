## 变量内存地址分析

```go
//
// Created by virus on 2022/5/20.
//

#include "stdio.h"

int main() {
    // <type> <name>
    int value; // 不赋值默认为0

    // <type> <name> = <initialized value>
    int value_init = 3;

    value = 4;
    value_init = 5;

    printf("value :%d\n", value);

    value_init = value;
    printf("size of value: %d\n", sizeof(value)); // 打印字节数

    printf("address of value: %#x\n", &value); // 打印十六进制内存地址
    return 0;
}
```
