## 函数的变长参数

```c
//
// Created by virus on 2022/5/27.
//

#include <stdio.h>
#include <stdarg.h>

void HandleVarargs(int arg_count, ...) {
    // 定义用于获取我们的变长参数
    va_list args;

    // 开始遍历
    va_start(args, arg_count);
    for (int i = 0; i < arg_count; ++i) {
        // 取出对应的参数
        int arg = va_arg(args, int);
        printf("%d: %d\n", i, arg);
    }

    // 结束遍历
    va_end(args);
}

int main(void) {

    HandleVarargs(4, 1, 2, 3, 4);
    return 0;
}
```

```bash
0: 1
1: 2
2: 3
3: 4
```
